    sysbench Install¶

    install 
    1. Install Sysbench
[root@localhost sysbench]# yum   --enablerepo=epel install sysbench
    create user

    • MySQL8以上の場合 

      -- MySQL8以上の場合
-- DB作成
> CREATE DATABASE sbtest;

-- ユーザー作成
> CREATE USER 'sbtest'@'localhost' IDENTIFIED BY 'sbtest';

-- ユーザーに全権限追加
> GRANT ALL PRIVILEGES ON `sbtest`.* TO 'sbtest'@'localhost' WITH GRANT OPTION;     

-- 権限の変更をデータベースに反映
> FLUSH PRIVILEGES;

    • MySQL8以下の場合 

      -- MySQL8以下の場合
mysql> CREATE DATABASE sbtest;
Query OK, 1 row affected (0.00 sec)

mysql> CREATE USER sbtest;
Query OK, 0 rows affected (0.00 sec)

mysql>  GRANT ALL PRIVILEGES ON `sbtest`.* TO 'sbtest'@'localhost' IDENTIFIED BY 'sbtest';
Query OK, 0 rows affected (0.00 sec)

mysql> FLUSH PRIVILEGES;
Query OK, 0 rows affected (0.00 sec)

    • 確認 

      -- 確認
> show databases;
> select user,host,authentication_string from mysql.user;

    • DBとユーザ削除時 

      # DBとユーザ削除時
> DROP DATABASE IF EXISTS `mydb`;
> select db,user FROM mysql.db WHERE user = 'myuser' ;
> DELETE FROM mysql.db WHERE user = 'myuser';
> DELETE FROM mysql.user WHERE user = 'myuser';

-- 削除後もう一度確認
> select db,user FROM mysql.db WHERE user = 'myuser' ;
> select user,host FROM mysql.user WHERE user = 'myuser' ;

    sysbench用スクリプト¶

    スクリプト 
    root@xxx ~]# cat sysbench_test.sh
#!/usr/bin/env bash
PATH=/bin:/sbin:/usr/bin:/usr/sbin:/usr/local/bin:/usr/local/sbin:~/bin
export PATH
set -e

clear
echo
echo "##############################################################################"
echo "# prepare:                                                                   #"
echo "# performs preparative actions for those tests which need them, e.g.         #"
echo "# creating the necessary files on disk for the fileio test,                  #"
echo "# or filling the test database for database benchmarks.                      #"
echo "# run:                                                                       #"
echo "# runs the actual test specified with the testname argument.                 #"
echo "# This command is provided by all tests.                                     #"
echo "# cleanup:                                                                   #"
echo "# removes temporary data after the test run in those tests which create one. #"
echo "# help:                                                                      #"
echo "# displays usage information for the test specified with the testname        #"
echo "# argument. This includes the full list of commands provided by the test,    #"
echo "# so it should be used to get the available commands.                        #"
echo "##############################################################################"
echo

THR=$1
CMD=$2

if [ ! "$THR" ]; then
  echo "Not threads set! "
  exit 1;
fi

if [ ! "$CMD" ]; then
  echo "Not command selected! "
  exit 1;
fi

sysbench oltp_read_write \
--mysql-dry-run=off \
--db-driver=mysql \
--mysql-socket=/tmp/mysql.sock \
--mysql-user=sbtest \
--mysql-password=sbtest \
--table_size=10000 \
--threads=$THR \
--tables=30 \
--time=120 \
--report-interval=10 \
$CMD

# オプション
1. table_size=10000 テーブルサイズ。デフォルトのまま
2. tebles=30 テーブル数。WPは基本的に29個テーブルのため、テストには30に設定
3. time=120 テスト時間。 2分間のテストを実施する
4. report-interval=10 ログの出力間隔。 10秒ごとレポートを出力する

# 使い方
1. 対象サーバ上DB, DB_USER 作成
CREATE DATABASE sbtest;
CREATE USER sbtest;
GRANT ALL PRIVILEGES ON sbtest.* TO 'sbtest'@'localhost' IDENTIFIED BY 'sbtest';
FLUSH PRIVILEGES;

>> MySQL8の場合
# CREATE USER 'sbtest'@'localhost' IDENTIFIED BY 'sbtest';
# GRANT all ON *.* TO 'sbtest'@'localhost';


2. テーブル準備
bash sysbench_test.sh 300 prepare

3. DBを温める
sysbench oltp_read_write --mysql-dry-run=off --db-driver=mysql --mysql-socket=/tmp/mysql.sock --mysql-user=sbtest --mysql-password=sbtest --table_size=10000 --threads=10 --tables=30 --time=60 --report-interval=10 run

3. 検証開始
- bash sysbench_test.sh 10 run
- for i in $(seq 10 10 300); do echo $i; bash sysbench_test.sh $i run >> sysbench.log; done

4. bash sysbench_test.sh 10 cleanup
# スクリプト引数
1. $1 は threads数、テスト時MySQLに同時接続数
2. ＄2 は command です、 順番として、まず、 prepare（準備）し、その後、run(実施)、最後に不要となるデータ削除cleanup。
