# DBとユーザ削除時
DROP DATABASE IF EXISTS `mydb`;
select db,user FROM mysql.db WHERE user = 'myuser' ;
DELETE FROM mysql.db WHERE user = 'myuser';
DELETE FROM mysql.user WHERE user = 'myuser';
select db,user FROM mysql.db WHERE user = 'myuser' ;
select user,host FROM mysql.user WHERE user = 'myuser' ;

    authentication_policyの指定がなければ、デフォルトの設定になります。¶

    authentication_policyを指定なし 
    # authentication_policyの指定がなければ、デフォルトの設定になります。
CREATE DATABASE `mydb`;
CREATE USER 'myuser'@'%' identified BY 'password';
GRANT ALL PRIVILEGES ON mydb.* TO 'myuser'@'%' WITH GRANT OPTION;
FLUSH PRIVILEGES;
select user,host,plugin from mysql.user where user = 'myuser';

    authentication_policyを指定してユーザの作成も可能です。¶

    authentication_policyを指定 
    # authentication_policyを指定してユーザの作成も可能です。
CREATE DATABASE `mydb`;
CREATE USER 'myuser'@'%' identified with caching_sha2_password BY 'password';
GRANT ALL PRIVILEGES ON mydb.* TO 'myuser'@'%' WITH GRANT OPTION;
FLUSH PRIVILEGES;
select user,host,plugin from mysql.user where user = 'myuser';
